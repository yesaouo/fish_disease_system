% ======================
% Stage 1: 病灶偵測
% ======================

\node[block] (input) {輸入魚隻影像};

    RF-DETR\\
    病灶偵測
};

\node[smallblock, left=22mm of detector] (healthy) {
    無 lesion\\
    判定健康
};

% ======================
% Stage 2: 證據抽取
% ======================

\node[block, below=of detector] (evidence) {
    Evidence Extraction\\
    產生多粒度證據
};

\node[smallblock, below left=8mm and 15mm of evidence] (global) {
    Global Token\\
    VLM-Global Vision
};

\node[smallblock, below=8mm of evidence] (lesion) {
    Lesion Tokens\\
    VLM-Lesion Fusion
};

\node[smallblock, below right=8mm and 15mm of evidence] (text) {
    Optional Text Token\\
    VLM-Global Text Enc.
};

\node[block, below=16mm of lesion] (queryset) {
    Query Evidence Set\\
    $Q=\{g_Q,\;t_Q,\;l_{Q1},\dots,l_{QN}\}$
};

% ======================
% Stage 3: 案例檢索 (C²R)
% ======================

\node[block, below=of queryset] (c2r) {
    C$^2$R Case Retrieval\\
    $\mathrm{sim}=\alpha\cos(g_Q,g_C)+\beta\,\mathrm{HungarianMatch}(L_Q,L_C)$\\
    {\scriptsize $\alpha=0.25,\;\beta=0.75$}
};

\node[block, below=of c2r] (candidates) {
    Top-$K$ Cases ($K{=}20$)\\
    $\rightarrow$ Candidate Cause Pool ($\sim\!87$)
};

% ======================
% Stage 4: CEAH v3 — Two-Stream Faithful Attribution
% ======================

\node[ceahblock, below=of candidates] (project) {
    Project all to common space\\
    {\scriptsize $g',\,t',\,l'_i,\,c'$ + type embedding}
};

\node[ceahblock, below=of project] (alpha) {
    $\alpha=\mathrm{softmax}(\mathrm{MLP}([c',e_i,c'\!\odot\! e_i]))$\\
    over $\{g,\,t,\,l_1,\dots,l_N\}$,\quad $\sum_i\alpha_i=1$
};

\node[ceahblock, below left=8mm and 5mm of alpha] (gstream) {
    Global Stream\\
    $z_g=\alpha_g\,e_g$\\
    $s_g=\sigma(\mathrm{MLP}_g([c',z_g]))$
};

\node[ceahblock, below right=8mm and 5mm of alpha] (lstream) {
    Local Stream\\
    $z_\ell=\sum_{i\in\{t,l\}}\alpha_i\,e_i$\\
    $s_\ell=\sigma(\mathrm{MLP}_\ell([c',z_\ell]))$
};

\node[ceahblock, below=40mm of alpha] (mult) {
    Multiplicative Score\\
    $s_{\mathrm{ceah}}(c)=s_g\cdot s_\ell$\\
    {\scriptsize 兩個 stream 都對才能高分}
};

% ======================
% Stage 5: Hybrid + Diversification + Output
% ======================

\node[block, below=of mult] (hybrid) {
    Hybrid Scoring\\
    $s(c)=\gamma\,s_{\mathrm{phase1}}(c)+(1{-}\gamma)\,s_{\mathrm{ceah}}(c)$\\
    {\scriptsize $\gamma=0.75$（per-query min-max norm）}
};

\node[block, below=of hybrid] (diversify) {
    Greedy Diversification\\
    {\scriptsize 若 $\max_{c'\in\mathrm{kept}}\cos(c,c')>0.95$ 則跳過}
};

\node[output, below=of diversify] (final) {
    最終輸出\\
    病因排序 + 支持病灶 + 支持案例
};

% ======================
% Right side: Case database + cause table
% ======================

\node[db, right=30mm of c2r] (db) {
    Case Database\\
    train cases
};

\node[smallblock, above=6mm of db, align=left, font=\footnotesize] (casecontent) {
    每筆 case：\\
    \quad global\_emb,\\
    \quad lesion\_embs[N],\\
    \quad text\_\{med,coll\}\_emb,\\
    \quad causes,\,cause\_emb\_idx
};

\node[db, right=90mm of project, fill=orange!8] (causedb) {
    cause\_text\_embs\\
    \scriptsize 56k cause emb table
};

% ======================
% Phase 1 candidate score (input to hybrid)
% ======================

\node[smallblock, left=22mm of hybrid, align=left, font=\footnotesize] (phase1score) {
    Phase 1 score\\

};

% ======================
